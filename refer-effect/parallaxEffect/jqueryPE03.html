.paraNav {
    top: -200px;
    transition: top 0.4s ease;
}
.paraNav.show {
    top: 20px;
}
.paraNav li a {
    width: auto;
    padding: 5px 20px;
}
.paraNav li.active a {
    border-radius: 30px;
}

$(".paraNav ul li").click(function(e){
    e.preventDefault();
    let target = $(this);
    let index = target.index();
    let section = $(".content-item").eq(index);
    let offset = section.offset().top;
    $("html, body").animate({ scrollTop : offset }, 600);
});

$(window).scroll(function(){
    let scrollTop = $(this).scrollTop();

    //숨김 메뉴
    // if(scrollTop > 10){
    //     $(".paraNav").addClass("show");
    // } else {
    //     $(".paraNav").removeClass("show");
    // }

    //한 줄 쓰기
    //(scrollTop > 10) ? $(".paraNav").addClass("show") : $(".paraNav").removeClass("show");

    //두 줄 쓰기
    if(scrollTop > 10) $(".paraNav").addClass("show");
    else $(".paraNav").removeClass("show");

    //버튼용
    $(".content-item").each(function(index){
       if(scrollTop >= $(this).offset().top - 2){
            $(".paraNav ul li").eq(index).addClass("active").siblings().removeClass("active");
       } 
    });

    //출력용
    $(".paraScroll span").text(parseInt(scrollTop));
});
