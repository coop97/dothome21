.cursor {
    position: absolute;
    left: 0; top: 0;
    width: 10px;
    height: 10px;
    border-radius: 50%;
    z-index: 9999;
    background-color: #fff;
    user-select: none;
    pointer-events: none;
    transition: transform 0.3s;
    mix-blend-mode: difference;

.cursor.active {
    transform: scale(20);
}
.info.left.list li.active {
    text-decoration: underline;
    color: rgb(111, 221, 216);

//출력용
$(".info.left.list li").click(function(){
    $(".info.left.list li").removeClass();
    $(this).addClass("active");
    const text = $(this).text();
    $(".cursor").css("mix-blend-mode", text);
});

//움직임 효과
const w = $(".cursor").width() / 2;
const h = $(".cursor").height() / 2;

$(window).mousemove(function(e){
    gsap.to(".cursor", {duration: 0.4, left: e.pageX - w, top: e.pageY - h});
});

//오버 효과
$(".mouseCont").hover(function(){
    $(".cursor").addClass("active");
}, function(){
    $(".cursor").removeClass("active")
});
