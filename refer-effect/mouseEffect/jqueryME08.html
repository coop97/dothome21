$(window).mousemove(function(e){
    const mouseX = e.clientX;
    const mouseY = e.clientY;

    $(".item").each(function(index, item){
        const itemX = $(this).offset().left + 25;
        const itemY = $(this).offset().top + 25;
        const diffX = mouseX - itemX;
        const diffY = mouseY - itemY;
        const radians = Math.atan2(diffY, diffX);
        const angle = radians * 180 / Math.PI;

        $(this).css({"transform": "rotate("+angle+"deg)"});
    });

    //커서를 원 가운데로
    const width = $(".cursor").width()/2;
    const height = $(".cursor").height()/2;

    //마우스 움직이기
    let pageX = e.pageX;
    let pageY = e.pageY;

    gsap.to(".cursor", {duration: 0.5, left: pageX -width, top: pageY -height});

    //출력
    $(".pageX").text(pageX);
    $(".pageY").text(pageY);
});
